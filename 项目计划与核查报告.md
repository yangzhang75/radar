### 1.2 ⚠️ 两项重要新发现（系统提示词 §3.1 未含，必须纳入设计）

1. **握手协议 `01B1`(请求链接) → `01B2`(允许链接)**：多播地址并非写死，雷达通过握手回报 **10 位序列号 + 雷达 IP + 完整协商后的多播地址清单**（文档给了完整字节示例）。这是连接建立的真实入口，直接影响 T1.1 连接流程与 T1.3。另：协议明确要求"自定义支持"——协议地址、命令头、10 位序列号均可写入自定义。
2. **上行看门狗 `A1C1`**：客户端必须每 ~8s 发送一次。超 ~30s 未发 → 雷达自动转待机；~1min → 关机。**通讯服务模块的心跳必须包含这条上行命令**（连同 `03C2`/`04C2`/`05C2` 周期查询，规则：A0C1+查询每 3s，状态 01C4 每 2s）。

## 3. 冻结接口契约草案（§3.6）— **待全员评审后冻结**

> 这是「接口先行」的核心产物。下面是建议的 Kotlin 数据模型；评审通过即冻结，之后才允许并行写依赖它的代码。所有角度统一约定：内部用 **度(Double, 0~360, 真北或船首参考由字段标注)**；原始 0~4095/0.1° 编码只存在于 comms 解析层，不外泄。

### 3.1 通讯 → 显示（输出数据流）
```kotlin
// 回波辐条帧（解析后极坐标）
data class EchoSpoke(
    val azimuthDeg: Double,          // 相对船首 0~360 (=360*spokeAzimuth/4096)
    val headingDeg: Double?,         // spokeCompass 解析；compassInvalid 时 null
    val trueNorth: Boolean,          // 真北/磁北
    val rangeCellSizeMm: Int,
    val rangeCellsDiv2: Int,
    val samples: ByteArray,          // 0..15/采样, 已解包成每元素1采样
    val encoding: SampleEncoding,    // AMPLITUDE / DOPPLER  ← T2.2 着色依据
    val sequenceNumber: Int,
    val bearingZeroError: Boolean
) { val maxRangeMeters: Double get() = (rangeCellSizeMm * 2.0 * rangeCellsDiv2) / 1000.0 }
enum class SampleEncoding { AMPLITUDE, DOPPLER }   // DOPPLER: 15=接近 14=远离

// 目标（雷达跟踪 / AIS 统一视图，来源用 source 区分）
data class TrackedTarget(
    val id: String, val source: TargetSource,       // RADAR_TT / AIS_ACTIVE / AIS_SLEEPING
    val rangeNm: Double, val bearingDeg: Double, val trueBearing: Boolean,
    val latitude: Double?, val longitude: Double?,
    val courseDeg: Double?, val speedKn: Double?,
    val cpaNm: Double?, val tcpaSec: Double?,
    val status: TargetStatus                          // ACQUIRING/TRACKED/LOST/TEST_MANEUVER...
)
enum class TargetSource { RADAR_TT, AIS_ACTIVE, AIS_SLEEPING }

// 本船
data class OwnShipData(
    val utc: Long?, val latitude: Double?, val longitude: Double?,
    val headingDeg: Double?, val headingTrue: Boolean,
    val cogDeg: Double?, val sogKn: Double?, val rotDegMin: Double?,
    val sourceValidity: Map<SensorKind, Boolean>      // 传感器失效指示(§3.7 必显)
)
enum class SensorKind { HEADING, POSITION, COG_SOG, SPEED_LOG, AIS, RADAR_LINK }

// 雷达状态回显（来自 01C4/02C4/08C4...）
data class RadarStatus(
    val powerState: RadarPowerState,                  // OFF/STANDBY/TRANSMIT/WARMUP/NO_SCANNER/DETECTING
    val rangeMeters: Int, val gainAuto: Boolean, val gain: Int,
    val seaMode: SeaMode, val seaLevel: Int, val rainLevel: Int,
    val interferenceRejection: Int, val targetExpansion: Boolean,
    val rpmX10: Int, val masterSlave: MasterSlave,
    val guardZones: List<GuardZoneStatus>, val warmupRemainSec: Int?
)

// 报警事件（BAM）
data class AlarmEvent(
    val identifier: Int,                              // 标准 ID: 3044/3052/3048/3042/3043/3015/3002...
    val priority: AlarmPriority,                      // ALARM/WARNING/CAUTION
    val state: AlarmState,                            // ACTIVE/ACK/RECTIFIED/SILENCED... (62923 状态机)
    val text: String?, val source: String, val utc: Long
)
```

### 3.2 显示 → 通讯（输入/控制命令）
```kotlin
sealed interface RadarCommand {
    data class Power(val on: Boolean) : RadarCommand           // 00C1
    data class Transmit(val on: Boolean) : RadarCommand        // 01C1
    data class SetRange(val meters: Int) : RadarCommand        // 03C1 (min 15)
    data class Gain(val auto: Boolean, val level: Int) : RadarCommand     // 06C1 type0
    data class Sea(val mode: SeaMode, val level: Int) : RadarCommand      // 06C1 type2
    data class Rain(val level: Int) : RadarCommand             // 06C1 type4
    data class Rpm(val rpm: Int) : RadarCommand                // 05CB (10~36)
    data class GuardZone(/* zone,startR,endR,bearing,width,onOff,alarmType,sensitivity */) : RadarCommand // 90C1
    // ...增益自动、干扰抑制、目标扩展等按 §1.3 逐条补齐
}
sealed interface TrackCommand {
    data class Acquire(val rangeNm: Double, val bearingDeg: Double) : TrackCommand
    data class Cancel(val targetId: String) : TrackCommand
}
```

### 3.3 总线接口（解耦的唯一通道）
```kotlin
interface RadarDataBus {                              // comms 实现，ui 只读订阅
    val echoSpokes: Flow<EchoSpoke>
    val targets: StateFlow<List<TrackedTarget>>
    val ownShip: StateFlow<OwnShipData>
    val radarStatus: StateFlow<RadarStatus>
    val alarms: Flow<AlarmEvent>
    val linkState: StateFlow<LinkState>               // 握手/连接/断线状态
}
interface RadarController {                            // ui 调用，comms 实现（内部编码下发组播）
    fun send(cmd: RadarCommand)
    fun send(cmd: TrackCommand)
}
```
> **冻结纪律**：本契约一旦评审通过，任何字段变更必须经编排者批准并全员同步（§6.3）。`shared/` 由编排者唯一维护，其余员工只读引用。
